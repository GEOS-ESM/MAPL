      subroutine CFIO_parseIntTime ( hhmmss, hour, min, sec )
         integer, intent(in)  :: hhmmss
         integer, intent(out) :: hour, min, sec
         hour = hhmmss / 10000
         min  = mod(hhmmss,10000)/100
         sec  = mod(hhmmss,100)
      end subroutine CFIO_parseIntTime
