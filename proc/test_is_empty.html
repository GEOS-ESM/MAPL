   subroutine test_is_empty()
      type (UnlimitedEntity) :: a
      logical :: is
      
      ! not initialized
      is = a%is_empty()
      @assertTrue(is)
     
      ! empty string
      a = UnlimitedEntity('')
      is = a%is_empty()
      @assertTrue(is)

      a = UnlimitedEntity('no')
      is = a%is_empty()
      @assertFalse(is)

      a = UnlimitedEntity(0)
      is = a%is_empty()
      @assertFalse(is)

      a = UnlimitedEntity([0])
      is = a%is_empty()
      @assertFalse(is)

   end subroutine test_is_empty
