   type :: CouplerMetaComponent

      type(ComponentDriverPtrVector) :: sources
      type(ComponentDriverVector) :: consumers
      logical :: stale = .true.

      procedure :: initialize
      procedure :: update
      procedure :: invalidate
      procedure :: clock_advance

      procedure, non_overridable :: is_stale
