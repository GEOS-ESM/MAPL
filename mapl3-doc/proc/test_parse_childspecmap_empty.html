   subroutine test_parse_ChildSpecMap_empty()
      type(ChildSpecMap) :: expected, found

      type(ESMF_HConfig) :: hconfig

      hconfig = ESMF_HConfigCreate(content='{}')

      found = parse_children(hconfig, _RC)
      @assert_that(found == expected, is(true()))
