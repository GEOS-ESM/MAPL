   subroutine test_is_active()
      type (AdvancedMeter) :: t

      t = AdvancedMeter(MpiTimerGauge())

      ! Initial state is
      @assertFalse(t%is_active())

      call t%start()
      @assertTrue(t%is_active())

      call t%stop()
      @assertFalse(t%is_active())

   end subroutine test_is_active
