   subroutine finalize(config, rc)
      type(ESMF_Config), intent(inout) :: config
      integer, optional, intent(out) :: rc

      integer :: status
      
      call MAPL_Finalize(_RC)
      call ESMF_ConfigDestroy(config, _RC)
      call ESMF_Finalize(_RC)

      _RETURN(_SUCCESS)
   end subroutine finalize
