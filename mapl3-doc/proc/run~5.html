   subroutine run(this, rc)
      class(BundleAction), intent(inout) :: this
      integer, optional, intent(out) :: rc
      
      integer :: i
      integer :: status
      class(ExtensionAction), pointer :: action
      
      do i = 1, this%actions%size()
         action => this%actions%of(i)
         call action%run(_RC)
      end do
      
      _RETURN(_SUCCESS)
   end subroutine run
