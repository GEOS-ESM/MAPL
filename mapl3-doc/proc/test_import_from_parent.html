   subroutine test_import_from_parent(this)
      class(ESMF_TestMethod), intent(inout) :: this
      type(HierarchicalRegistry), target :: r_parent, r_child
      type(VirtualConnectionPt) :: vpt_parent, vpt_child
      type(SimpleConnection) :: conn
      integer :: status
      type(ActualPtVector), pointer :: actual_pts

      r_parent = HierarchicalRegistry('parent')
      r_child = HierarchicalRegistry('child')
      call r_parent%add_subregistry(r_child)

      vpt_parent = new_v_pt('export', 'ae')
      vpt_child = new_v_pt('import', 'ai')

      call r_parent%add_item_spec(vpt_parent, MockItemSpec('AE'))
      call r_child%add_item_spec(vpt_child, MockItemSpec('AI'))

      conn = SimpleConnection(CP('parent', vpt_parent), CP('child', vpt_child))
      call r_parent%add_connection(conn, rc=status)
      @assert_that(status, is(0))

!!$      if (.not. check(r_child, vpt_child, ['AE'])) return

      ! Check that extension was created
      actual_pts => r_parent%get_actual_pts(vpt_parent)
      @assert_that(int(actual_pts%size()), is(2))
   end subroutine test_import_from_parent
