   subroutine run_geos(config, rc)
      type(ESMF_Config), intent(inout) :: config
      integer, optional, intent(out) :: rc

      type(ESMF_HConfig) :: cap_hconfig
      integer :: status

      call ESMF_ConfigGet(config, hconfig=cap_hconfig, _RC)
