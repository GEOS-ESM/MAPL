   subroutine test_e2e_preserve_actual_pt()
      type(HierarchicalRegistry), target :: r
      type(HierarchicalRegistry), target :: r_A
      type(VirtualConnectionPt) :: vpt_1, vpt_2

      integer :: status

      r = HierarchicalRegistry('R')
      r_A = HierarchicalRegistry('A')
      call r%add_subregistry(r_A)

      vpt_1 = new_v_pt('export', 'ae1')
      vpt_2 = new_v_pt('export', 'ae2')

      ! True export
      call r_A%add_item_spec(vpt_1, MockItemSpec('AE1'))

      ! E-to-E with rename
      call r%add_connection(ReexportConnection(CP('A',vpt_1), CP('R',vpt_2)), rc=status)

      @assert_that(r%has_item_spec(ActualConnectionPt(vpt_2)), is(true()))

   end subroutine test_e2e_preserve_actual_pt
