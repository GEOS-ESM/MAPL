   logical function match_geom(a, b) result(match)
      type(ESMF_Geom), allocatable, intent(in) :: a, b

      integer :: status

      match = .false.
      
      if (allocated(a) .and. allocated(b)) then
         match = MAPL_SameGeom(a, b)
      end if

   end function match_geom
