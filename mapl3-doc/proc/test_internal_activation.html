   subroutine test_internal_activation()
      type(HierarchicalRegistry) :: r
      class(StateItemSpec), pointer :: spec

      type(ActualConnectionPt) :: apt_1, apt_2, apt_3
           apt_1 = new_a_pt('internal', 'A')
           apt_2 = new_a_pt('export', 'A')
           apt_3 = new_a_pt('import', 'A')

        call r%add_item_spec(apt_1, MockItemSpec('A1'))
        call r%add_item_spec(apt_2, MockItemSpec('A2'))
        call r%add_item_spec(apt_3, MockItemSpec('A3'))

        spec => r%get_item_spec(apt_1)
        @assert_that(spec%is_active(), is(true()))
        
        spec => r%get_item_spec(apt_2)
        @assert_that(spec%is_active(), is(false()))
        
        spec => r%get_item_spec(apt_3)
        @assert_that(spec%is_active(), is(false()))
        
   end subroutine test_internal_activation
