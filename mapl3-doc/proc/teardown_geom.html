   subroutine teardown_geom(geom)
      type(ESMF_Geom), intent(inout) :: geom

      type(ESMF_Grid) :: grid

      call ESMF_GeomGet(geom, grid=grid)
      call ESMF_GridDestroy(grid)
      call ESMF_GeomDestroy(geom)

   end subroutine teardown_geom
