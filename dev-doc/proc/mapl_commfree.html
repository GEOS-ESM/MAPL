  subroutine MAPL_CommFree(comm, rc)
    integer, intent(inout) :: comm
    integer, optional, intent(out) :: rc

    integer :: status

    if(comm /= MPI_COMM_NULL) then
       call MPI_Comm_Free(comm, status)
       _VERIFY(status)
       comm = MPI_COMM_NULL
    end if
    _RETURN(ESMF_SUCCESS)
  end subroutine MAPL_CommFree
