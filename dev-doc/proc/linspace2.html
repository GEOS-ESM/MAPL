SUBROUTINE linspace2(d1,d2,n,y)
        IMPLICIT   NONE
        REAL(dp), INTENT(IN)   :: d1, d2
        INTEGER, INTENT(IN)   :: n
        INTEGER            :: n2, i
        REAL(dp), DIMENSION(:), INTENT(INOUT) :: y

        n2 = n
        DO i = 0, n2-2
          y(i+1) = i 
        END DO
        y(1:n2-1) = (d1+y(1:n2-1)*(d2-d1)/(n-1))   ! d2);
        y(n2) = d2
END SUBROUTINE linspace2
