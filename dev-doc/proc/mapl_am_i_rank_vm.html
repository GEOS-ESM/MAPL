  function MAPL_Am_I_Rank_Vm(VM, rank) result(R)
    type (ESMF_VM)    :: VM
    integer, optional :: rank
    logical           :: R

    integer        :: deId
    integer        :: status
    integer        :: rank_

    rank_ = MAPL_Root
    if (present(rank)) rank_ = rank

    call ESMF_VMGet(VM, localPet=deId, rc=status)
    R = .false.
    if (deId == rank_) R = .true.

  end function MAPL_Am_I_Rank_Vm
