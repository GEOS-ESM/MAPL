   integer function get_length(this) result(length)
      class (ReplaceMetadataMessage), intent(in) :: this
      integer, allocatable :: buffer(:)

      allocate(buffer(0))

      call this%fmd%serialize(buffer)

      length = &
           & serialize_buffer_length(this%collection_id) + &
           & size(buffer)
