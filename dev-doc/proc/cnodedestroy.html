   subroutine cNodeDestroy(List)
     type(cNode), pointer :: List, p, q

     if (.not. associated(List)) return
     q => List
     p => List%next
     do while ( associated(p) )  
        if (associated(q)) then
           deallocate(q)
        end if
        q => p
        p => p%next
     end do

     if (associated(q)) then 
        deallocate(q)
     end if

   end subroutine cNodeDestroy
