  subroutine parse_timeunit_i4(tunit, n, t0, dt, rc)
    use ESMF

    character(len=*), intent(in) :: tunit
    integer, intent(in) :: n
    type(ESMF_Time), intent(out) :: t0
    type(ESMF_TimeInterval), intent(out) :: dt
    integer, optional, intent(out) :: rc
    integer :: status
    integer(ESMF_KIND_I8) :: n8

    n8 = n

   _RETURN(_SUCCESS)
