   subroutine MAPL_VarSpecDestroy1(SPEC, RC )

      type (MAPL_VarSpec ),             pointer      :: SPEC(:)
      integer            , optional   , intent(OUT)  :: RC



      integer :: i

      if (associated(SPEC)) then
         do I=1,size(SPEC)
            call MAPL_VarSpecDestroy0(spec(i))
         end do
         deallocate(SPEC)
      end if
