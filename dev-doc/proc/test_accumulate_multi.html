   subroutine test_accumulate_multi()
      type(TimeProfiler), target :: main, lap

      main = TimeProfiler('main')
      call main%start()
      lap = TimeProfiler('lap')
      call lap%start()
      call lap%start('A')
      call lap%stop('A')
      call lap%finalize()
      call main%accumulate(lap)

      call lap%reset()
      call lap%start('A')
      call lap%stop('A')
      call lap%finalize()
      call main%accumulate(lap)
      
      
   end subroutine test_accumulate_multi
