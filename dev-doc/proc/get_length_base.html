   integer function get_length_base(this) result(length)
      class (AbstractDataReference), intent(in) :: this
      integer :: n
      
      if( allocated(this%shape)) then
         n = size(this%shape)
      else
         n = 0
      endif

      length = size(transfer(this%base_address,[1])) +  &
               1 + 1 + & ! type_kind + rank 
               n  ! shape

   end function get_length_base
