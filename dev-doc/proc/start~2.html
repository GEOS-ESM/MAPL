   subroutine start(this)
      class(AdvancedMeter), intent(inout) :: this

      !$omp master
      if (this%active) then
         this%status = MAPL_METER_START_ACTIVE
      else
         this%active = .true.
         this%start_value = this%gauge%get_measurement()
      end if
      !$omp end master

   end subroutine start
