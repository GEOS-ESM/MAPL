   function is_time_unit(tunit)
      character(len=*), intent(in) :: tunit
      logical :: is_time_unit
      integer :: i

      is_time_unit = .TRUE.
      do i = 1, size(TIME_UNITS)
         if(lr_trim(tunit) == lr_trim(TIME_UNITS(i))) return
      end do
      is_time_unit = .FALSE.

   end function is_time_unit
