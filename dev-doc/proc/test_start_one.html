   subroutine test_start_one()
      type (TimeProfiler), target :: prof
      
      prof = TimeProfiler('top')
      call Prof%start()

      call prof%start('timer_1')
      call prof%stop('timer_1')

      call prof%finalize()
      
      @assertEqual(2, prof%get_num_meters())
      
   end subroutine test_start_one
